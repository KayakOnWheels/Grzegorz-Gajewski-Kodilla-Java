SELECT IL.NAME, I.SUMMARY, I.DESCRIPTION
FROM issueslists IL
JOIN issues I ON IL.ID = I.ISSUESLIST_ID;

SELECT I.SUMMARY, IL.NAME, U.FIRSTNAME, U.LASTNAME
FROM issueslists IL
JOIN issues I ON IL.ID = I.ISSUESLIST_ID
JOIN users U ON I.USER_ID_ASSIGNEDTO = U.ID;

SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) AS TASKS_COUNT
FROM ISSUES I
JOIN USERS U ON I.USER_ID_ASSIGNEDTO = U.ID
GROUP BY I.USER_ID_ASSIGNEDTO
